function getShipCells(r, c, size, dir) {
  const cells = [];
  for (let i = 0; i < size; i++) {
    const nr = dir === 'V' ? r + i : r;
    const nc = dir === 'H' ? c + i : c;
    if (nr >= 10 || nc >= 10) return null;
    cells.push({ r: nr, c: nc, i: idx(nr, nc) });
  }
  return cells;
}

const pCell = (r, c) => document.querySelector(`#pRows .cell[data-r="${r}"][data-c="${c}"]`);

function clearPreview() {
  document.querySelectorAll('#pRows .cell.preview').forEach(el => el.classList.remove('preview'));
}

function previewShip(r, c) {
  clearPreview();
  if (!selectedShip) return;
  const s     = SHIPS.find(x => x.id === selectedShip);
  const cells = getShipCells(r, c, s.size, orient);
  if (!cells) return;
  cells.forEach(({ r: nr, c: nc }) => { const el = pCell(nr, nc); if (el) el.classList.add('preview'); });
}

function placeAt(r, c) {
  if (!selectedShip) { toast('Select a ship first', 'i'); return; }
  const s     = SHIPS.find(x => x.id === selectedShip);
  const cells = getShipCells(r, c, s.size, orient);
  if (!cells) { toast('Out of bounds', 'e'); return; }
  if (cells.some(({ i }) => myBoard[i])) { toast('Overlapping another ship', 'e'); return; }
  cells.forEach(({ i }) => { myBoard[i] = s.id; });
  placedShips[s.id] = cells.map(({ i }) => i);
  cells.forEach(({ r: nr, c: nc }) => {
    const el = pCell(nr, nc);
    if (el) { el.classList.remove('preview'); el.classList.add('placed'); }
  });
  document.getElementById('si_' + s.id).className = 'ship-item done';
  selectedShip = null;
  clearPreview();
  if (Object.keys(placedShips).length === SHIPS.length) {
    document.getElementById('btnReady').disabled = false;
    toast('All ships placed — click Ready!', 's');
  }
}

async function pollGame() {
  if (!roomId || !EMAIL) return;
  const state = await api(`/battle/state?room_id=${roomId}&email=${encodeURIComponent(EMAIL)}`);
  if (!state || state.error) return;
  handleState(state);
}

function handleState(s) {
  if (s.status === 'waiting' || s.status === 'placing') {
    show('vWait');
    return;
  }
  if (s.status === 'active' || s.status === 'finished') {
    const inGame = document.getElementById('vGame').classList.contains('active');
    if (!inGame && s.status === 'active') toast('Battle started!', 's');
    show('vGame');
    renderBoards(s.my_board, s.enemy_board, s.my_turn && s.status === 'active');
    updateStatus(s);
    appendLog(s.move_log || []);
    if (s.status === 'finished' && !gameOver) {
      gameOver = true;
      clearInterval(pollTimer);
      showEnd(s.winner === EMAIL);
    }
  }
}

function buildGrid(colId, rowId, data, clickable, onFire) {
  document.getElementById(colId).innerHTML = COLS.map(c => `<div class="cl">${c}</div>`).join('');
  const wrap = document.getElementById(rowId);
  wrap.innerHTML = '';
  for (let r = 0; r < 10; r++) {
    const row = document.createElement('div');
    row.className = 'grid-row-wrap';
    row.innerHTML = `<div class="rl">${ROWS[r]}</div>`;
    for (let c = 0; c < 10; c++) {
      const cell = document.createElement('div');
      const val  = data ? data[r * 10 + c] : 0;
      cell.className = 'cell';
      if      (val === 1 || val === 'miss') cell.classList.add('miss');
      else if (val === 2 || val === 'hit')  cell.classList.add('hit');
      else if (val === 3 || val === 'sunk') cell.classList.add('sunk');
      else if (val === 'ship')              cell.classList.add('placed');
      else if (clickable && val === 0)      cell.classList.add('target');
      if (clickable && val === 0) cell.addEventListener('click', () => onFire(r, c));
      row.appendChild(cell);
    }
    wrap.appendChild(row);
  }
}

function appendLog(log) {
  if (log.length === lastLogLen) return;
  const el = document.getElementById('moveLog');
  for (let i = lastLogLen; i < log.length; i++) {
    const e   = log[i];
    const cls = {hit:'lh', miss:'lm', sunk:'ls'}[e.result] || '';
    const ico = {hit:'💥', miss:'💧', sunk:'🔥'}[e.result] || '·';
    el.innerHTML += `<span class="${cls}">${ico} ${esc(e.actor)} → ${COLS[e.col]}${ROWS[e.row]} — ${e.result}</span>\n`;
  }
  lastLogLen = log.length;
  el.scrollTop = el.scrollHeight;
}

// Exit spectator mode
function stopSpectating() {
  clearInterval(specPollTimer);
  isSpectating = false;
  specRoomId   = null;
  specGameOver = false;
  specLastLogLen = 0;
  document.getElementById('specEndModal').classList.remove('open');

// Spectator poll loop (2s)
function startSpecPoll() {
  clearInterval(specPollTimer);
  specPollTimer = setInterval(pollSpectator, 2000);
  pollSpectator();
}

async function pollSpectator() {
  if (!specRoomId) return;
  // Use the dedicated spectator endpoint — returns both boards fully revealed
  const state = await api(`/battle/spectate?room_id=${encodeURIComponent(specRoomId)}`);
  if (!state || state.error) {
    // Room gone — show finished notice and bounce back
    if (!specGameOver) stopSpectating();
    return;
  }
  handleSpecState(state);
}

// Build a read-only grid for spectator — ships are revealed with a purple tint
// data is flat 100-element array using same encoding as /battle/state but
// also includes 'ship' for live ship positions so spectators can see the fleet
function buildSpecGrid(colId, rowId, data) {
  document.getElementById(colId).innerHTML = COLS.map(c => `<div class="cl">${c}</div>`).join('');
  const wrap = document.getElementById(rowId);
  wrap.innerHTML = '';
  for (let r = 0; r < 10; r++) {
    const row = document.createElement('div');
    row.className = 'grid-row-wrap';
    row.innerHTML = `<div class="rl">${ROWS[r]}</div>`;
    for (let c = 0; c < 10; c++) {
      const cell = document.createElement('div');
      const val  = data ? data[r * 10 + c] : null;
      cell.className = 'cell';
      // Spectator gets to see ships too — special purple tint for intact cells
      if      (val === 'ship')  cell.classList.add('ship-spec');
      else if (val === 'miss')  cell.classList.add('miss');
      else if (val === 'hit')   cell.classList.add('hit');
      else if (val === 'sunk')  cell.classList.add('sunk');
      row.appendChild(cell);
    }
    wrap.appendChild(row);
  }
}

function handleSpecState(s) {
  // If game ended
  if (s.status === 'finished' && !specGameOver) {
    specGameOver = true;
    clearInterval(specPollTimer);
    // Do a final render then show modal
    renderSpecBoards(s);
    const winner = s.winner_email || null;
    const hostEmail   = document.getElementById('specBoard1Name').textContent;
    const guestEmail  = document.getElementById('specBoard2Name').textContent;
    document.getElementById('specEndTitle').textContent = '⚓ Battle Over';
    document.getElementById('specEndMsg').textContent   = winner
      ? `${winner === hostEmail ? hostEmail : guestEmail} wins by sinking the enemy fleet!`
      : 'The game has ended.';
    document.getElementById('specEndModal').classList.add('open');
    return;
  }

  // If the room disappeared (abandoned etc.) while we're watching
  if (s.status === 'abandoned' || s.status === 'cancelled') {
    clearInterval(specPollTimer);
    toast('Room was abandoned', 'e');
    stopSpectating();
    return;
  }

  renderSpecBoards(s);
}

function renderSpecBoards(s) {
  // Board A = host, Board B = guest
  // Worker returns host_board and guest_board (both with ships visible)
  buildSpecGrid('sACols', 'sARows', s.host_board);
  buildSpecGrid('sBCols', 'sBRows', s.guest_board);

  // Ships left
  document.getElementById('specALeft').textContent = `${s.host_ships_left ?? '?'} ships`;
  document.getElementById('specBLeft').textContent = `${s.guest_ships_left ?? '?'} ships`;

  // Turn indicator
  const hostEmail  = document.getElementById('specBoard1Name').textContent;
  const isHostTurn = s.turn_email === s.host_email;

  document.getElementById('specTurnArrow1').style.display = isHostTurn  ? '' : 'none';
  document.getElementById('specTurnArrow2').style.display = !isHostTurn ? '' : 'none';

  // Status bar
  const tl = document.getElementById('specTurnLabel');
  if (s.status === 'finished') {
    tl.textContent = `🏆 ${s.winner_email || 'Game over'} wins!`;
    tl.className   = 'turn-label mine';
  } else {
    tl.textContent = `🎯 ${s.turn_email || '?'}'s turn`;
    tl.className   = 'turn-label spec';
  }
  document.getElementById('specShipCounts').textContent =
    `${hostEmail}: ${s.host_ships_left ?? '?'} ships · ${document.getElementById('specBoard2Name').textContent}: ${s.guest_ships_left ?? '?'} ships`;

  // Move log (append only)
  appendSpecLog(s.move_log || []);
}

function appendSpecLog(log) {
  if (log.length === specLastLogLen) return;
  const el = document.getElementById('specMoveLog');
  for (let i = specLastLogLen; i < log.length; i++) {
    const e   = log[i];
    const cls = {hit:'lh', miss:'lm', sunk:'ls'}[e.result] || '';
    const ico = {hit:'💥', miss:'💧', sunk:'🔥'}[e.result] || '·';
    el.innerHTML += `<span class="${cls}">${ico} ${esc(e.actor)} → ${COLS[e.col]}${ROWS[e.row]} — ${e.result}</span>\n`;
  }
  specLastLogLen = log.length;

}
